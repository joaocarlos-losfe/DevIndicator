softwrares para o desenvolvimento do projeto

dados necessarios para criar uma conta
{
    email,
    senha,
    nome de usuario
    {
        funcionará como id unico,
        será adicionado automaticamente no post 
    }
}

dados para necessarios para o post
{
    nome do usuario que fez o post
    {
        necessario o usuario tá logado,
        capturar automaticamente (não exibir ao precher e não permitir alteração)
    }

    data da postagem
    {
        adicionar automaticamente no momento do post,
        terá dia, mes, ano e horario (ex: 23/02/2022 14:45)
    },
    
    categoria
    {
        no frontend, as categorias disponiveis devem vir do backend,
        deverá ser incluido automaticamente nas tags,
    },

    tags
    {
        não obrigatorio, mais melhora o mecanismo de busca por posts, 
    }

    titulo
    {
        funcionará como id unico (não sera permitido ter outro projeto como o mesmo titulo),
        incluir automaticamente nas tags (será utilizado nas buscas)
    }

    post
    {
        texto detalhado contendo uma descrição do post (limitar a quantidade de caracteres no front)  
    }

    url do projeto
    {
        funcionará como id unico (não será permitido ter outro projeto com o mesmo link)
    }

    utilizar o uuid
    {
	    gerar um id para o post como um todo
    }

}

dados necessarios para logar na conta
{
    email,
    senha
}
